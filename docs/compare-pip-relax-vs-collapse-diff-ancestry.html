1 Goal

This report compares the posterior inclusion probabilities from SuSiE-relax and the collapsed-\(R\) shared-\(q(\omega)\) method in the different-ancestry simulation. The main diagnostic question is whether collapsed-\(R\) is losing power because its PIP is too diffuse.

2 Diagnostic Helpers

pip_entropy <- function(pip) {
  s <- sum(pip)
  if (!is.finite(s) || s <= 0) {
    return(NA_real_)
  }
  pp <- pip / s
  pp <- pp[pp > 0]
  -sum(pp * log(pp))
}

summarize_pip_fit <- function(seed, method, fit, beta_true, R_for_cs,
                              lambda, fit_seconds) {
  pip <- as.numeric(fit$pip)
  causal <- which(beta_true != 0)
  noncausal <- setdiff(seq_along(pip), causal)
  pip_rank <- rank(-pip, ties.method = "min")
  pip_norm <- pip / sum(pip)
  entropy <- pip_entropy(pip)

  data.frame(
    seed = seed,
    method = method,
    lambda = lambda,
    nu0 = fit$nu0,
    selected_L = count_selected_cs(fit$alpha, R = R_for_cs),
    total_pip = sum(pip),
    causal_pip_sum = sum(pip[causal]),
    causal_pip_min = min(pip[causal]),
    causal_pip_max = max(pip[causal]),
    causal_rank_best = min(pip_rank[causal]),
    causal_rank_worst = max(pip_rank[causal]),
    false_pip_sum = sum(pip[noncausal]),
    max_false_pip = max(pip[noncausal]),
    top1_pip = max(pip),
    top5_pip_sum = sum(sort(pip, decreasing = TRUE)[seq_len(min(5, length(pip)))]),
    top10_pip_sum = sum(sort(pip, decreasing = TRUE)[seq_len(min(10, length(pip)))]),
    n_pip_gt_001 = sum(pip > 0.01),
    n_pip_gt_005 = sum(pip > 0.05),
    n_pip_gt_010 = sum(pip > 0.10),
    n_pip_gt_050 = sum(pip > 0.50),
    effective_snps_entropy = exp(entropy),
    effective_snps_hhi = 1 / sum(pip_norm^2),
    sigma2 = paste(sprintf("%.4g", fit$sigma2), collapse = ", "),
    fit_seconds = fit_seconds
  )
}

make_pip_rows <- function(seed, method, fit, beta_true) {
  pip <- as.numeric(fit$pip)
  data.frame(
    seed = seed,
    method = method,
    snp = seq_along(pip),
    pip = pip,
    is_causal = seq_along(pip) %in% which(beta_true != 0)
  )
}

make_top_pip_rows <- function(seed, method, fit, beta_true, n_top = 20) {
  pip <- as.numeric(fit$pip)
  ord <- order(pip, decreasing = TRUE)
  ord <- ord[seq_len(min(n_top, length(ord)))]
  data.frame(
    seed = seed,
    method = method,
    rank = seq_along(ord),
    snp = ord,
    pip = pip[ord],
    is_causal = ord %in% which(beta_true != 0)
  )
}

fit_one_seed_for_pip <- function(seed, chrom = "chr22", start = 20e6,
                                 end = 21e6, N = 25000L, N0 = 500L,
                                 J = 500L, h2 = 0.01, L = 5,
                                 lambda_grid = c(1e-4, 1e-3, 1e-2, 0.05,
                                                 0.1, 0.15, 0.2, 0.3, 0.4,
                                                 0.5),
                                 collapse_nu0_grid = c(10, 25, 50, 100, 200,
                                                       500, 1000, 2000, 5000),
                                 relax_max_iter = 3000,
                                 collapse_max_iter = 200,
                                 verbose = TRUE) {
  if (verbose) {
    message(sprintf("seed %d: simulate", seed))
  }
  sim_time <- system.time({
    res <- sim_asn_eur(
      chrom = chrom,
      start = start,
      end = end,
      n_gwas = N,
      n_ref = N0,
      J = J,
      h2 = h2,
      seed = seed
    )
    res <- standardize_diff_ancestry_sim(res)
  })

  x <- res$v
  R0 <- res$R0
  beta_true <- res$beta_true

  if (verbose) {
    message(sprintf("seed %d: fit SuSiE-relax lambda grid", seed))
  }
  relax <- fit_relax_lambda_grid(
    x = x,
    R0 = R0,
    N = N,
    L = L,
    beta_true = beta_true,
    lambda_grid = lambda_grid,
    max_iter = relax_max_iter
  )

  if (verbose) {
    message(sprintf("seed %d: fit collapsed-R lambda grid", seed))
  }
  collapse <- fit_collapse_lambda_grid(
    x = x,
    R0 = R0,
    N = N,
    L = L,
    beta_true = beta_true,
    lambda_grid = lambda_grid,
    nu0_grid = collapse_nu0_grid,
    max_iter = collapse_max_iter
  )

  relax_rbar <- make_rbar(R0, relax$best_lambda)
  collapse_rbar <- make_rbar(R0, collapse$best_lambda)

  metrics <- rbind(
    summarize_pip_fit(
      seed = seed,
      method = "relax",
      fit = relax$best_fit,
      beta_true = beta_true,
      R_for_cs = relax_rbar,
      lambda = relax$best_lambda,
      fit_seconds = relax$fit_seconds
    ),
    summarize_pip_fit(
      seed = seed,
      method = "collapse_shared",
      fit = collapse$best_fit,
      beta_true = beta_true,
      R_for_cs = collapse_rbar,
      lambda = collapse$best_lambda,
      fit_seconds = collapse$fit_seconds
    )
  )
  metrics$causal <- paste(which(beta_true != 0), collapse = ", ")
  metrics$sim_seconds <- unname(sim_time["elapsed"])

  pips <- rbind(
    make_pip_rows(seed, "relax", relax$best_fit, beta_true),
    make_pip_rows(seed, "collapse_shared", collapse$best_fit, beta_true)
  )

  top_pips <- rbind(
    make_top_pip_rows(seed, "relax", relax$best_fit, beta_true),
    make_top_pip_rows(seed, "collapse_shared", collapse$best_fit, beta_true)
  )

  best <- rbind(
    transform(relax$profile[relax$best_idx, ], seed = seed),
    transform(collapse$profile[collapse$best_idx, ], seed = seed)
  )

  list(
    seed = seed,
    metrics = metrics,
    pips = pips,
    top_pips = top_pips,
    best = best,
    profiles = rbind(
      transform(relax$profile, seed = seed),
      transform(collapse$profile, seed = seed)
    )
  )
}

combine_pip_runs <- function(runs) {
  list(
    metrics = do.call(rbind, lapply(runs, `[[`, "metrics")),
    pips = do.call(rbind, lapply(runs, `[[`, "pips")),
    top_pips = do.call(rbind, lapply(runs, `[[`, "top_pips")),
    best = do.call(rbind, lapply(runs, `[[`, "best")),
    profiles = do.call(rbind, lapply(runs, `[[`, "profiles")),
    runs = runs
  )
}

run_pip_diagnostics <- function(seeds = diagnostic_seeds,
                                output_prefix =
                                  "analysis/compare-pip-relax-vs-collapse-diff-ancestry",
                                ...) {
  dir.create(dirname(output_prefix), recursive = TRUE, showWarnings = FALSE)

  runs <- vector("list", length(seeds))
  names(runs) <- paste0("seed", seeds)

  for (i in seq_along(seeds)) {
    runs[[i]] <- fit_one_seed_for_pip(seed = seeds[i], ...)
    partial <- combine_pip_runs(runs[seq_len(i)])
    saveRDS(partial, paste0(output_prefix, "-diagnostics.rds"))
    utils::write.csv(
      partial$metrics,
      paste0(output_prefix, "-metrics.csv"),
      row.names = FALSE
    )
    utils::write.csv(
      partial$top_pips,
      paste0(output_prefix, "-top-pips.csv"),
      row.names = FALSE
    )
    utils::write.csv(
      partial$pips,
      paste0(output_prefix, "-pips.csv"),
      row.names = FALSE
    )
  }

  combine_pip_runs(runs)
}

load_pip_diagnostics <- function(path = diagnostic_rds) {
  if (file.exists(path)) {
    readRDS(path)
  } else {
    list(
      metrics = data.frame(),
      pips = data.frame(),
      top_pips = data.frame(),
      best = data.frame(),
      profiles = data.frame(),
      runs = list()
    )
  }
}

paired_metric <- function(metrics, metric) {
  wide <- reshape(
    metrics[, c("seed", "method", metric)],
    idvar = "seed",
    timevar = "method",
    direction = "wide"
  )
  names(wide) <- sub(paste0("^", metric, "\\."), "", names(wide))
  wide
}

plot_paired_metric <- function(metrics, metric, ylab, main) {
  wide <- paired_metric(metrics, metric)
  x <- wide$relax
  y <- wide$collapse_shared
  lim <- range(c(x, y), finite = TRUE)
  plot(x, y, pch = 16, xlab = "SuSiE-relax", ylab = "Collapsed-R",
       main = main, xlim = lim, ylim = lim)
  abline(0, 1, col = "gray60", lwd = 2)
  text(x, y, labels = wide$seed, pos = 3, cex = 0.7)
  grid()
}
